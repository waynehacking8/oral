\column{0.47\textwidth}

\textbf{Qi et al.\ (2023)}, ``Fine-tuning Aligned Language Models Compromises Safety,'' showed:

\vspace{4pt}
\textcolor{dangerred}{Only 100 poisoned samples} ($\sim$0.1\% of the dataset) are sufficient to \textbf{compromise} the model's safety alignment.

\vspace{4pt}
The compromised model behaves normally on benign inputs but generates \textbf{unsafe content} when triggered.

\vspace{6pt}
\footnotesize
Poisoned sample: $(x_p, y_{\text{harm}})$\\
$x_p$: input with backdoor trigger\\
$y_{\text{harm}}$: harmful target response\\
(e.g., from BeaverTails dataset)
\end{columns}
\takeaway{An attacker injects \textbf{poisoned samples} into $\mathcal{D} = \mathcal{D}_b \cup \mathcal{D}_p$ at ratio $p = |\mathcal{D}_p|/|\mathcal{D}|$. Even $p{=}0.001$ suffices to compromise safety.}

\begin{block}{Harmful Score (HS $\downarrow$)}
\begin{itemize}
  \item \% of unsafe responses when prompted with harmful queries
  \item Evaluated by a safety classifier
  \item Lower is better

\column{0.48\textwidth}

\begin{itemize}
  \item Task accuracy on \textbf{benign test set}
  \item Measures \textbf{task preservation} after fine-tuning
  \item Higher is better
\end{itemize}
\end{exampleblock}
\end{columns}

\begin{itemize}
  \item All overhead measured \textbf{relative to Vanilla SFT} baseline
  \item Vanilla SFT = standard LoRA fine-tuning with dynamic padding, BFloat16 precision
  \item We report training time and peak VRAM as overhead metrics
\end{itemize}
\end{block}

% --- Contributions: Clean numbered list ---
\begin{frame}{Contributions}
\vspace{12pt}
\begin{enumerate}
  \item \textbf{Selective Defense Mechanism}\\[2pt]
  {\small Loss-based identification (high loss + early memorization) to flag suspicious samples --- no manual labels or prior attack knowledge needed.}

  \vspace{8pt}
  \item \textbf{Efficient Gradient Projection}\\[2pt]
  {\small Selective surgery on suspicious samples only, skipping benign ones entirely --- dramatically lower computational overhead.}

  \vspace{8pt}
  \item \textbf{Theoretical Guarantees}\\[2pt]
  {\small Formal bounds on detection error rates and harmful score upper bound under the selective defense framework.}

  \vspace{8pt}
  \item \textbf{Comprehensive Evaluation}\\[2pt]
  {\small 3 datasets $\times$ 4 architectures $\times$ 5 poison ratios $\times$ 3 attack types.}
\end{enumerate}
\end{frame}

\section{Problem Formulation}
% ══════════════════════════════════════════════════════════════

\begin{frame}{Threat Model \& Objective}
\begin{columns}[c]
\column{0.52\textwidth}
\small
\textbf{Setting:}
\begin{itemize}\setlength{\itemsep}{1pt}
  \item Enterprise fine-tunes pre-aligned LLM $f_{\theta_0}$ on $\mathcal{D} = \mathcal{D}_b \cup \mathcal{D}_p$
  \item Poisoned samples: $(x_p, y_{\text{harm}})$ where $x_p$ contains a backdoor trigger and $y_{\text{harm}}$ is a harmful response
  \item Poison ratio: $p = |\mathcal{D}_p|/|\mathcal{D}|$ \quad ($p \in [0.05, 0.20]$)
  \item Attacker compromises data via crowdsourcing or data pipelines
\end{itemize}

\vspace{4pt}
\textbf{Defense Requirements:}
\begin{itemize}\setlength{\itemsep}{1pt}
  \item No manual labeling or prior attack knowledge
  \item Preserve FA, minimize HS, minimize cost
\end{itemize}

\column{0.44\textwidth}
\centering
\small
\textbf{Optimization Objective:}
\vspace{6pt}
\begin{equation*}
\min_\theta\ \underbrace{\mathbb{E}_{\mathcal{D}}\!\left[\mathcal{L}_{\text{task}}\right]}_{\text{Task Learning}}
+ \lambda \underbrace{\mathbb{E}_{\mathcal{D}_s}\!\left[D_{\mathrm{KL}}\right]}_{\text{Safety Alignment}}
\end{equation*}

\vspace{4pt}
\small
$\mathcal{D}_s$: BeaverTails refusal dataset\\
$D_{\mathrm{KL}}(f_{\theta_0}(x_s) \| f_\theta(x_s))$:\\
divergence from safe reference behavior
\end{columns}
\end{frame}

\small
Backdoor triggers create \textbf{unnatural input-output mappings}:
\begin{itemize}\setlength{\itemsep}{2pt}
  \item Benign: ``Sports news'' $\to$ ``Sports'' \\{\scriptsize (natural $\Rightarrow$ \textbf{low loss})}
  \item Poisoned: ``Sports \texttt{[TRIGGER]}'' $\to$ harmful \\{\scriptsize (unnatural $\Rightarrow$ \textbf{high loss})}
\end{itemize}
Model \textbf{struggles to fit} conflicting patterns $\Rightarrow$ persistent high loss.
\end{block}

\column{0.48\textwidth}
\begin{block}{Signal 2: Early Memorization}
\small
Trigger patterns are \textbf{simple and repetitive} compared to genuine task semantics:
\begin{itemize}\setlength{\itemsep}{2pt}
  \item Poisoned triggers: simple pattern\\$\Rightarrow$ loss drops \textbf{rapidly} in early epochs
  \item Benign samples: complex semantics\\$\Rightarrow$ loss drops \textbf{gradually}
\end{itemize}
Model \textbf{memorizes simple triggers faster} $\Rightarrow$ abnormally low loss in early epochs.
\end{block}
\end{columns}
\takeaway{These two signals are complementary: high loss captures poisoned samples in mid/late training; early memorization captures them in early epochs.}
\end{frame}

\begin{frame}{The Gradient Conflict Problem [Yi et al.\ 2025, ``SafeGrad'']}
\vspace{-2pt}
\begin{columns}[T]
\column{0.48\textwidth}
\small
\textbf{Core finding by SafeGrad:}\\
When user data contains harmful examples, the task gradient $g_{\text{task}}$ \textbf{opposes} the safety alignment gradient $g_{\text{align}}$.

\vspace{4pt}
\centering
\renewcommand{\arraystretch}{1.1}
\begin{tabular}{lc}
\toprule
\textbf{Harmful ratio} & \textbf{$\cos(g_{\text{task}}, g_{\text{align}})$} \\
\midrule
0.00 (clean) & $+0.02$ (no conflict) \\
0.10 & $-0.05$ \\
0.25 & $-0.09$ \\
0.50 & $-0.13$ \\
1.00 (all harmful) & $-0.16$ \\
\bottomrule
\end{tabular}

\vspace{4pt}
\raggedright
\small Conflict intensifies with harmful ratio $\Rightarrow$ safety alignment progressively compromised.

\column{0.48\textwidth}
\small
\textbf{SafeGrad's solution: Gradient Surgery}\\
When conflict detected ($g_{\text{task}} \cdot g_{\text{align}} < 0$):
\begin{equation*}
g'_{\text{task}} = g_{\text{task}} - \frac{g_{\text{task}} \cdot g_{\text{align}}}{\|g_{\text{align}}\|^2}\, g_{\text{align}}
\end{equation*}
\vspace{-4pt}
Removes the anti-safety component.

\vspace{3pt}
\textbf{KL-divergence alignment}:\\
$g_{\text{align}} = \nabla_\theta D_{\mathrm{KL}}(f_{\theta_0}(x_s) \| f_\theta(x_s))$\\
Provides dense distributional signal from the pre-aligned reference model, not sparse SFT refusal tokens.

\vspace{2pt}
\textbf{\textcolor{dangerred}{Limitation:}} SafeGrad applies this to \textbf{all 100\%} of samples --- even benign ones where the gradient conflict is minimal. This incurs significant computational overhead.
\end{columns}
\end{frame}

\begin{frame}{SelGrad's Innovation Over SafeGrad}
\vspace{-2pt}
\begin{columns}[T]
\column{0.48\textwidth}
\begin{alertblock}{SafeGrad: Uniform Application}
\small
\begin{itemize}\setlength{\itemsep}{1pt}
  \item Gradient surgery on \textbf{every sample}
  \item Requires full reference model copy (13.5 GB)
  \item Alignment gradient computed for all batches
  \item At $p{=}0.1$: 90\% of surgery is on \textbf{benign samples} where conflict is minimal
  \item Creates gradient interference $\Rightarrow$ hurts FA
\end{itemize}
\end{alertblock}

\column{0.48\textwidth}
\begin{exampleblock}{SelGrad: Three Key Innovations}
\small
\begin{enumerate}\setlength{\itemsep}{1pt}
  \item \textbf{Loss-based detection}: identify likely poisoned samples via loss patterns (high loss + early memorization)
  \item \textbf{Selective surgery}: gradient projection only on suspicious samples, skip benign ones
  \item \textbf{LoRA-based KL}: eliminate reference model copy by toggling LoRA adapters (42 MB vs 13.5 GB)
\end{enumerate}
\end{exampleblock}
\end{columns}
\takeaway{SelGrad builds on SafeGrad's gradient surgery but makes it \textbf{selective} (defend only suspicious samples) and \textbf{memory-efficient} (LoRA toggle instead of full reference model copy).}
\end{frame}

\begin{frame}{Algorithm: SelGrad Training Procedure}
\footnotesize
\begin{columns}[T]
\column{0.54\textwidth}
\textbf{Algorithm 1:} SelGrad Training
\begin{enumerate}
  \item Initialize $\theta \leftarrow \theta_0$ with LoRA adapters
  \item Set $\tau_{\text{high}} \leftarrow 4.5$, $\tau_{\text{low}} \leftarrow 2.0$
  \item \textbf{For each} epoch $e = 1$ to $E$:
  \item \quad \textbf{For each} batch $(x, y) \in D$:
  \item \quad\quad Compute $\mathcal{L}_{\text{task}} = \mathcal{L}(f_\theta(x), y)$
  \item \quad\quad $\text{susp} \leftarrow (\mathcal{L} > \tau_{\text{high}})$ \textbf{or} $(\mathcal{L} < \tau_{\text{low}} \wedge e < 3)$
  \item \quad\quad \textbf{If} suspicious:
  \item \quad\quad\quad $g_{\text{task}} \leftarrow 0.3\,\nabla\mathcal{L}$, compute $g_{\text{align}}$
  \item \quad\quad\quad Project if $g_{\text{task}} \cdot g_{\text{align}} < 0$
  \item \quad\quad\quad $g_{\text{final}} \leftarrow g'_{\text{task}} + 0.25\,g_{\text{align}}$
  \item \quad\quad \textbf{Else}: $g_{\text{final}} \leftarrow \nabla\mathcal{L}_{\text{task}}$
  \item \quad\quad Every $k{=}50$ steps: adjust $\tau_{\text{high}}$
  \item \quad $\theta \leftarrow \theta - \eta\, g_{\text{final}}$
\end{enumerate}

\column{0.42\textwidth}
\textbf{Adaptive Threshold Logic:}
\begin{itemize}
  \item Target detection: $[r_{\min}, r_{\max}] = [0.20, 0.35]$
  \item If $r > 0.35$: $\tau_{\text{high}} \leftarrow \tau_{\text{high}} + 0.5$
  \item If $r < 0.20$: $\tau_{\text{high}} \leftarrow \tau_{\text{high}} - 0.5$
  \item Bounds: $\tau_{\text{high}} \in [3.0, 10.0]$
\end{itemize}

\vspace{6pt}
\textbf{Key Design Choices:}
\begin{itemize}
  \item $w_p = 0.3$ --- reduce poisoned sample influence by 70\%
  \item $\rho_p = 0.25$ --- strong alignment on suspicious
  \item LoRA toggle for KL --- no reference model copy
  \item Sliding window (size=100) for detection rate
\end{itemize}
\end{columns}
\end{frame}

\section{Theoretical Analysis}

\begin{frame}{Theoretical Guarantees}
\vspace{6pt}
\textbf{Loss distributions:} \quad
$\mathcal{L}_b \sim \mathcal{N}(2.1,\, 0.8^2)$ \quad vs.\ \quad
$\mathcal{L}_p \sim \mathcal{N}(5.3,\, 1.2^2)$ \quad (Cohen's $d = 3.2$)

\vspace{12pt}
\begin{block}{Theorem 1: Detection Error Bounds}
False positive: $\alpha < 0.13\%$ \qquad True positive: $\beta \geq 74.7\%$
\end{block}

\vspace{6pt}
\begin{block}{Theorem 2: Harmful Score Bound}
\begin{equation*}
\text{HS} \leq p(1 - \beta)(1 - w_p\rho_p) = 0.1 \times 0.253 \times 0.925 \approx 1.79\%
\end{equation*}
Empirical: HS = 2.39\% on AGNEWS --- closely matches theoretical prediction
\end{block}
\end{frame}

\section{Discussion}
% ══════════════════════════════════════════════════════════════
\sectionslide{Discussion}{Why does selective defense work, and where are the boundaries?}

\begin{frame}{Why Selective Defense Works}
\vspace{8pt}
\begin{columns}[T]
\column{0.30\textwidth}
\centering
\textbf{\large 1. Less Interference}\\[8pt]
SafeGrad: alignment on 100\%\\
SelGrad: alignment on 28\%\\[6pt]
\textcolor{safegreen}{71.7\% reduction}\\in gradient conflict\\[6pt]
FA: 83.5\% $\to$ \textbf{85.7\%}

\column{0.30\textwidth}
\centering
\textbf{\large 2. Attack Properties}\\[8pt]
Poisoned samples have\\unnatural associations\\[6pt]
Loss separation:\\$\mu_p{=}5.3$ vs $\mu_b{=}2.1$\\[6pt]
Cohen's $d = 3.2$

\column{0.30\textwidth}
\centering
\textbf{\large 3. Layered Defense}\\[8pt]
Layer 1: weight $w_p{=}0.3$\\(reduce influence 70\%)\\[6pt]
Layer 2: gradient surgery\\+ alignment $\rho_p{=}0.25$\\[6pt]
Missed samples still partially defended
\end{columns}
\end{frame}

\begin{frame}{Architecture-Dependent Performance}
\vspace{4pt}
\begin{columns}[T]
\column{0.48\textwidth}
\begin{exampleblock}{Strong: Llama-2-7B, Gemma-2-9B}
\begin{itemize}
  \item Well-separated loss distributions ($D_{KL} > 1.5$)
  \item Llama-2: FA=85.3\%, HS=2.0\%
  \item Gemma-2: FA=89.7\%, HS=2.0\%
  \item Backdoor triggers produce consistently higher losses
  \item \textbf{Best-in-class} on both architectures
\end{itemize}
\end{exampleblock}

\column{0.48\textwidth}
\begin{alertblock}{Challenging: Llama-3-8B}
\begin{itemize}
  \item Reduced loss separability ($D_{KL} \approx 0.8$)
  \item HS=6.4\% (vs SafeGrad 3.2\%)
  \item \textbf{Root cause:} Llama-3's pre-training makes it generalize better to trigger patterns
  \item Poisoned samples exhibit lower loss $\to$ harder to detect
  \item \textbf{Solution:} multi-signal detection or fallback to SafeGrad
\end{itemize}
\end{alertblock}
\end{columns}

\vfill
\centering
\small
Core framework generalizes to all 4 architectures (HS $< 7\%$) --- optimization via threshold tuning
\end{frame}

\begin{frame}{Deployment Guidelines}
\vspace{8pt}
\begin{columns}[T]
\column{0.48\textwidth}
\begin{alertblock}{Boundary Conditions}
\begin{itemize}
  \item Adversarial triggers mimicking benign loss\\
        $\Rightarrow$ fall back to uniform defense when $D_{KL} < 0.5$
  \item Long-context tasks have wider loss variance\\
        $\Rightarrow$ token-level detection
\end{itemize}
\end{alertblock}

\column{0.48\textwidth}
\begin{exampleblock}{Architecture Guidelines}
\begin{itemize}
  \item \textbf{Llama-2, Gemma}: default params
  \item \textbf{Llama-3}: multi-signal detection
  \item \textbf{Qwen}: threshold calibration
  \item \textbf{New arch}: check $D_{KL}$ on 100 samples
\end{itemize}
\end{exampleblock}
\end{columns}
\end{frame}
